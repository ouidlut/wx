--#  DESCRIPTION    :    BEFORE行トリガー
--#                      T001＿BIZ_CUSTにDML によって変更後のデータを履歴として保存
--#  CHANGE HISTORY
--#   ----------    --------------------------------------------------    ------------
--#   DATE            DESCRIPTION                                             AUTHOR
--#   ----------    --------------------------------------------------    ------------
--#   2009/01/27    新規作成                                              YY.ZHANG
--#   2010/12/02    項目追加、フォーマット                                YG.PANG
--#

PROCEDURE BIZ_CUST_LOG (P_CHANE_TYPE VARCHAR2, NEW_BIZ_CUST IN OUT NOCOPY T001_BIZ_CUST%ROWTYPE, OLD_BIZ_CUST IN OUT NOCOPY T001_BIZ_CUST%ROWTYPE)
IS
    --#　INSERTを'I'、 DELETEを'D'、UPDATEを'U'で操作タイプを表示する
    V_CHANE_TYPE VARCHAR2(1) := P_CHANE_TYPE;
BEGIN
    --#　削除の場合、削除の前のデータを保存
    CASE
    WHEN V_CHANE_TYPE = 'D' THEN
        INSERT INTO T003_BIZ_CUST_LOG
            (
                 BIZ_CUST_LOG_ID        --# 変更されたデータを記録するID
            ,    LOG_INS_DT             --# 更新時刻
            ,    LOG_OPR_FLG            --# 操作フラグ
            ,    BIZ_CUST_ID            --# 業務別顧客ID：Hibernate参照
            ,    BIZ_ID                 --# 事業ID：M002事業マスタのPK参照
            ,    CUST_ID                --# 顧客ID：M001顧客マスタのPK参照
            )
        VALUES
            (
                 SEQ_T003_BIZ_CUST_LOG.NEXTVAL
            ,    CURRENT_TIMESTAMP
            ,    V_CHANE_TYPE
            ,    OLD_BIZ_CUST.BIZ_CUST_ID
            ,    OLD_BIZ_CUST.BIZ_ID
            ,    OLD_BIZ_CUST.CUST_ID
            );
    WHEN V_CHANE_TYPE = 'U' OR V_CHANE_TYPE = 'I' THEN
        INSERT INTO T003_BIZ_CUST_LOG
            (
                 BIZ_CUST_LOG_ID         --変更されたデータを記録するID
            ,    LOG_INS_DT              --更新時刻
            ,    LOG_OPR_FLG             --操作フラグ
            ,    BIZ_CUST_ID             --業務別顧客ID
            ,    INS_USER_ID             --登録ユーザーID
            ,    INS_DT                  --登録日時
            ,    UPD_USER_ID             --修正ユーザーID
            ,    UPD_DT                  --修正日時
            ,    BIZ_ID                  --事業ID
            ,    CUST_ID                 --顧客ID
            ,    ADDR_ID                 --住所ID
            ,    ADDR_PREF_CD            --住所コード（都道府県）
            ,    ADDR_CITY_CD            --住所コード（群市区）
            ,    ADDR_ZONE_CD            --住所コード（町村）
            ,    CONTACT_TEL_NO          --連絡先電話番号
            ,    FIND_TEL_NO             --検索電話番号
            ,    CALL_RST_ID             --最新コール結果
            ,    CALL_RST_RANK_CD        --最新ランク区分
            ,    CALL_DT                 --最新コール日時
            ,    APO_DT                  --客先指定日時
            ,    CALL_USER_ID            --営業員ID
            ,    CALL_USER_NM            --営業員氏名
            ,    GRP_ID                  --グループ
            ,    SUB_GRP_ID              --サブグループID
            ,    LIST_ID                 --リストID
            ,    LIST_TYPE_ID            --リストタイプID

IS

    V_BIZ_ID T001_BIZ_CUST.BIZ_ID%TYPE := NEW_BIZ_CUST.BIZ_ID;

    CASE
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_NOCALL)       --# 未コール
            THEN NEW_BIZ_CUST.EDIT_FLG :='00zz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_OTHERS)       --# その他
            THEN NEW_BIZ_CUST.EDIT_FLG :='00zz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_ABSENCE)      --# 留守
            THEN NEW_BIZ_CUST.EDIT_FLG :='00zz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_EXPECT)       --# 見込み
            THEN NEW_BIZ_CUST.EDIT_FLG :='066z066000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_FAX)          --# FAX
            THEN NEW_BIZ_CUST.EDIT_FLG :='0zzz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_NO_KEY_MAN)   --# 現担当不在
            THEN NEW_BIZ_CUST.EDIT_FLG :='00zz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_GEN_ANA)      --# 現アナ
            THEN NEW_BIZ_CUST.EDIT_FLG :='0zzz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_REJECT)       --# 拒否
            THEN NEW_BIZ_CUST.EDIT_FLG :='0zzz000000';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_CLAIM)        --# クレーム
            THEN NEW_BIZ_CUST.EDIT_FLG :='0zzz0zzz00';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_ORDER)        --# 受注成立
            THEN NEW_BIZ_CUST.EDIT_FLG :='06z6066088';
        WHEN NEW_BIZ_CUST.CALL_RST_ID IN (PKG000_COMMON.C_RST_APO_BAN)      --# アポ禁
            THEN NEW_BIZ_CUST.EDIT_FLG :='0zzz0zzz00';

--#====================================================================================================
--#  OBJECT NAME   :  USER_CALL_STAT
--#  DESCRIPTION   :  個人集計（コール数一覧）
--#
--#  CHANGE HISTORY
--#   ----------    --------------------------------------------------    ------------
--#   DATE            DESCRIPTION                                             AUTHOR
--#   ----------    --------------------------------------------------    ------------
--#   2009/01/31    新規作成                                              F.YANG
--#
--#====================================================================================================
PROCEDURE USER_CALL_STAT (P_CHANGE_TYPE VARCHAR2, NEW_BIZ_CUST IN OUT NOCOPY T001_BIZ_CUST%ROWTYPE, OLD_BIZ_CUST IN OUT NOCOPY T001_BIZ_CUST%ROWTYPE)
IS
    V_CHANGE_TYPE T003_BIZ_CUST_LOG.LOG_OPR_FLG%TYPE := P_CHANGE_TYPE;
    V_BIZ_ID T001_BIZ_CUST.BIZ_ID%TYPE := NEW_BIZ_CUST.BIZ_ID;
    V_CUST_ID T001_BIZ_CUST.CUST_ID%TYPE := NEW_BIZ_CUST.CUST_ID;
    C_SYS_ADMIN_ID T001_BIZ_CUST.CALL_USER_ID%TYPE := 'tmadmin';

BEGIN
    NULL;
    IF V_CHANGE_TYPE = 'U'

        AND NEW_BIZ_CUST.UPD_USER_ID <> C_SYS_ADMIN_ID 
        AND NEW_BIZ_CUST.CALL_DT IS NOT NULL
        AND (
                NEW_BIZ_CUST.CALL_USER_ID <> NVL(OLD_BIZ_CUST.CALL_USER_ID,' ')
            OR  NEW_BIZ_CUST.CALL_RST_ID <> NVL(OLD_BIZ_CUST.CALL_RST_ID,' ') 
            OR  NEW_BIZ_CUST.CALL_RST_RANK_CD <> NVL(OLD_BIZ_CUST.CALL_RST_RANK_CD,' ')
            OR  NEW_BIZ_CUST.CALL_DT <> OLD_BIZ_CUST.CALL_DT
            OR    (
                    NEW_BIZ_CUST.CALL_DT IS NOT NULL AND OLD_BIZ_CUST.CALL_DT IS NULL
                )
            ) 
    THEN
        MERGE INTO S018_USER_CALL_STAT S018
        USING (SELECT 
                    TRUNC(NEW_BIZ_CUST.CALL_DT) CALL_DT,
                    TO_CHAR(NEW_BIZ_CUST.CALL_DT, 'HH24') CALL_HOUR,
                    NEW_BIZ_CUST.BIZ_ID BIZ_ID, 
                    NEW_BIZ_CUST.CALL_USER_ID CALL_USER_ID, 
                    NEW_BIZ_CUST.CALL_RST_ID CALL_RST_ID,
                    NVL(NEW_BIZ_CUST.CALL_RST_RANK_CD, ' ') CALL_RST_RANK_CD
                FROM DUAL) T001_NEW
        ON (
                S018.CALL_DT = T001_NEW.CALL_DT
            AND S018.CALL_HOUR = T001_NEW.CALL_HOUR
            AND S018.BIZ_ID = T001_NEW.BIZ_ID
            AND S018.USER_ID = T001_NEW.CALL_USER_ID
            AND S018.CALL_RST_ID = T001_NEW.CALL_RST_ID
            AND NVL(S018.CALL_RST_RANK_CD,' ') = T001_NEW.CALL_RST_RANK_CD
            )
        WHEN MATCHED THEN
            UPDATE SET S018.CALL_CNT = S018.CALL_CNT + 1
            ,	S018.UPD_USER_ID = PKG000_COMMON.C_BATCH_USER_ID
            ,	S018.UPD_DT = SYSDATE
        WHEN NOT MATCHED THEN
            INSERT (
                USER_CALL_STAT_ID,
                INS_USER_ID,
                INS_DT,
                UPD_USER_ID,
                UPD_DT,
                CALL_DT,
                CALL_HOUR,
                BIZ_ID,
                USER_ID,
                CALL_RST_ID,
                CALL_RST_RANK_CD,
                CALL_CNT)
            VALUES(
                SEQ_S018_USER_CALL_STAT_ID.NEXTVAL ,     -- USER_CALL_STAT_ID
                NEW_BIZ_CUST.UPD_USER_ID ,                -- INS_USER_ID
                SYSDATE ,                                -- INS_DT
                NEW_BIZ_CUST.UPD_USER_ID ,                -- UPD_USER_ID
                SYSDATE ,                                -- UPD_DT
                T001_NEW.CALL_DT ,                        -- CALL_DT
                T001_NEW.CALL_HOUR ,                    -- CALL_HOUR
                T001_NEW.BIZ_ID ,                        -- BIZ_ID
                T001_NEW.CALL_USER_ID ,                        -- USER_ID
                T001_NEW.CALL_RST_ID ,                    -- CALL_RST_ID
                T001_NEW.CALL_RST_RANK_CD ,                -- CALL_RST_RANK_CD
                1 );                                    -- CALL_CNT
        
        --# 見込み、再コール、アポ成立の場合、通話者の情報を更新・挿入する
        /*
        	S018には下記値で更新・挿入
        	決裁者または社長:	-6
        	決裁者以外:			-7
        	会話せず:			-8
        */
		IF NEW_BIZ_CUST.CALL_RST_ID IN( PKG000_COMMON.C_RST_EXPECT, PKG000_COMMON.C_RST_RE_CALL , PKG000_COMMON.C_RST_APO_OK ) AND NEW_BIZ_CUST.RESV9 IN( '1', '2', '3' , '4', '5', '6', '7', '8') THEN
        	MERGE INTO S018_USER_CALL_STAT S018
	        USING (SELECT 
	                    TRUNC(NEW_BIZ_CUST.CALL_DT) CALL_DT,
	                    TO_CHAR(NEW_BIZ_CUST.CALL_DT, 'HH24') CALL_HOUR,
	                    NEW_BIZ_CUST.BIZ_ID BIZ_ID, 
	                    NEW_BIZ_CUST.CALL_USER_ID CALL_USER_ID, 
						CASE NEW_BIZ_CUST.RESV9
	                    	WHEN '1' THEN '-6'
	                    	WHEN '5' THEN '-6'
	                    	WHEN '8' THEN '-8'
				ELSE '-7'
	                    END CALL_RST_ID
	                FROM DUAL) T001_NEW
	        ON (
	                S018.CALL_DT = T001_NEW.CALL_DT
	            AND S018.CALL_HOUR = T001_NEW.CALL_HOUR
	            AND S018.BIZ_ID = T001_NEW.BIZ_ID
	            AND S018.USER_ID = T001_NEW.CALL_USER_ID
	            AND S018.CALL_RST_ID = T001_NEW.CALL_RST_ID
	            )
	        WHEN MATCHED THEN
	            UPDATE SET
	            	S018.CALL_CNT = S018.CALL_CNT + 1
	            ,	S018.UPD_USER_ID = PKG000_COMMON.C_BATCH_USER_ID
            	,	S018.UPD_DT = SYSDATE
	        WHEN NOT MATCHED THEN
	            INSERT (
	                USER_CALL_STAT_ID,
	                INS_USER_ID,
	                INS_DT,
	                UPD_USER_ID,
	                UPD_DT,
	                CALL_DT,
	                CALL_HOUR,
	                BIZ_ID,
	                USER_ID,
	                CALL_RST_ID,
	                CALL_RST_RANK_CD,
	                CALL_CNT)
	            VALUES(
	                SEQ_S018_USER_CALL_STAT_ID.NEXTVAL ,     -- USER_CALL_STAT_ID
	                NEW_BIZ_CUST.UPD_USER_ID ,                -- INS_USER_ID
	                SYSDATE ,                                -- INS_DT
	                NEW_BIZ_CUST.UPD_USER_ID ,                -- UPD_USER_ID
	                SYSDATE ,                                -- UPD_DT
	                T001_NEW.CALL_DT ,                        -- CALL_DT
	                T001_NEW.CALL_HOUR ,                    -- CALL_HOUR
	                T001_NEW.BIZ_ID ,                        -- BIZ_ID
	                T001_NEW.CALL_USER_ID ,                        -- USER_ID
	                T001_NEW.CALL_RST_ID ,                    -- CALL_RST_ID
	                NULL ,                -- CALL_RST_RANK_CD
	                1 );                                    -- CALL_CNT
		END IF;
    END IF;
END;
